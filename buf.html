@def(define buf)
class Buf {
std::string _value;
std::string _file;
int _startLine;
int _endLine;
bool _active;
@expand(private methods);
public:
@expand(buf methods);
};
@end(define buf)
@def(buf methods)
Buf(): _active(false) {}
@end(buf methods)
@add(buf methods)
bool active() const {
return _active;
}
@end(buf methods)
@add(buf methods)
bool empty() const {
return _value.empty();
}
@end(buf methods)
@add(buf methods)
void activate() {
_active = true;
}
@end(buf methods)
@add(buf methods)
void clear(bool active = false) {
_value.clear();
_file.clear();
_startLine = 0;
_endLine = 0;
_active = active;
}
@end(buf methods)
@add(buf methods)
const std::string &str() const {
return _value;
}
@end(buf methods)
@add(buf methods)
const std::string &file() const {
return _file;
}
@end(buf methods)
@add(buf methods)
int startLine() const {
return _startLine;
}
@end(buf methods)
@add(buf methods)
int endLine() const {
return _endLine;
}
@end(buf methods)
@def(private methods)
bool _canContinue(
const std::string &file,
int line
) {
if (_file != file) {
return false;
}
return
line == _endLine ||
line == _endLine + 1;
}
@end(private methods)
@add(buf methods)
bool canContinue(
const std::string &file,
int line
) {
if (_file.empty()) {
return true;
}
return _canContinue(
file, line
);
}
@end(buf methods)
@add(private methods)
void assertCont(
const std::string &file,
int line
) {
if (_file.empty()) {
_file = file;
_startLine = line;
_endLine = line;
}
ASSERT(_canContinue(file, line));
}
@end(private methods)
@add(buf methods)
void add(
const std::string &value,
const std::string &file =
"/dev/null",
int line = 1
) {
activate();
if (value.empty()) { return; }
assertCont(file, line);
_value += value;
_endLine = line;
}
@end(buf methods)
@add(buf methods)
void add(
int ch,
const std::string &file =
"/dev/null",
int line = 1
) {
char c = static_cast<char>(ch);
add(
std::string(&c, &c + 1),
file, line
);
}
@end(buf methods)
@add(buf methods)
void add(const Buf &b) {
activate();
if (b.empty()) { return; }
assertCont(
b._file, b._startLine
);
_value += b._value;
_endLine = b._endLine;
}
@end(buf methods)
