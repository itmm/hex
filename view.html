@Add(run loop)
if (cmd == "q" || cmd == "quit") {

}
@End(run loop)
@Add(run loop)
if (cmd == "n" || cmd == "next") {
if (curInput != inputs.end()) {
if (curBlock != curInput->blocks.end()) {
++curBlock;
draw_block();
continue;
}
}
std::cerr << "! end\n";
}
@End(run loop)
@Add(run loop)
if (cmd == "p" || cmd == "prev") {
if (curInput != inputs.end()) {
if (curBlock != curInput->blocks.begin()) {
--curBlock;
draw_block();
continue;
}
}
std::cerr << "! start\n";
}
@End(run loop)
