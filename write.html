@def(needed by write_x)
std::string split(
std::string &s, int width
) {
auto b { s.begin() };
auto e { s.end() };
while (b != e && *b == ' ') { ++b; }
auto c = b;
while (c != e) {
auto t = c;
while (t != e && *t == ' ') { ++t; }
while (t != e && *t != ' ') { ++t; }
if (c == b || t - b <= width) {
c = t;
} else {
break;
}
}
std::string res { b, c };
s.erase(s.begin(), c);
return res;
}
@end(needed by write_x)

void multi_write(
std::ofstream &out,
std::string str,
std::string first_in,
const std::string &other_in
) {
while (! str.empty()) {
std::string p = split(str, 72 - first_in.size());
out << first_in << p << '\n';
first_in = other_in;
}
}
